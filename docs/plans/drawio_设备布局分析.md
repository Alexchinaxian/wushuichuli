### 6.3 坐标转换方案

```csharp
// 建议的坐标转换逻辑
public static CoordinatesConfig ConvertFromDrawio(string name, double drawioX, double drawioY, 
                                                 double drawioWidth, double drawioHeight)
{
    // 整体偏移量，使布局居中于Canvas
    const double offsetX = 50;
    const double offsetY = 50;
    const double scale = 0.8; // 可选缩放系数
    
    return new CoordinatesConfig
    {
        ElementId = GenerateId(name),
        ElementType = ClassifyType(name),
        X = drawioX * scale + offsetX,
        Y = drawioY * scale + offsetY,
        Width = drawioWidth * scale,
        Height = drawioHeight * scale,
        Comment = $"来自drawio: {name}",
        IsVisible = true
    };
}
```
